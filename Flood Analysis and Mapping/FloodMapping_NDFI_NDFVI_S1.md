      function refinemap(img){
        // Include JRC layer on surface water seasonality to mask flood pixels from areas
        // of &quot;permanent&quot; water (where there is water &gt; 10 months of the year)
        var swater = ee.Image('JRC/GSW1_0/GlobalSurfaceWater').select('seasonality');
        var swater_mask = swater.gte(10).updateMask(swater.gte(10));
      
        //Flooded layer where perennial water bodies (water &gt; 10 mo/yr) is assigned a 0 value
        var flooded_mask = img.where(swater_mask,0);
        // var flooded_mask = difference_binary;
        // final flooded area without pixels in perennial waterbodies
        var flooded = flooded_mask.updateMask(flooded_mask);
      
        // Compute connectivity of pixels to eliminate those connected to 8 or fewer neighbours
        // This operation reduces noise of the flood extent product 
        var connections = flooded.connectedPixelCount();    
        var flooded = flooded.updateMask(connections.gte(8));
    
        // Mask out areas with more than 5 percent slope using a Digital Elevation Model 
        var DEM = ee.Image('WWF/HydroSHEDS/03VFDEM');
        var terrain = ee.Algorithms.Terrain(DEM);
        var slope = terrain.select('slope');
        var flooded = flooded.updateMask(slope.lt(5));
        return flooded
      }
